Đây là mã SQL được dùng để tạo 3 Views trong BigQuery (nằm trong file loaders/bigquery_loader.py của bạn).

Bạn sẽ thấy rằng các View này thực chất là việc dùng JOIN và các hàm kết hợp (SUM, COUNT, CASE WHEN, COALESCE...) để nối các bảng fact và dim lại với nhau thành một bảng phẳng (flat table) chứa đầy đủ thông tin, sẵn sàng cho việc kéo thả trên Power BI.

1. View: Báo cáo Dòng tiền Hàng ngày (vw_cashflow_daily)
View này được tạo ra bằng cách gom (group) dữ liệu theo ngày từ 3 bảng: fact_orders (đơn hàng), fact_payments (thanh toán online), và fact_bank_transactions (tiền ngân hàng nhận được), sau đó dùng FULL OUTER JOIN kết nối chúng lại theo cột ngày (report_date).

Nhờ bảng này, Dashboard Dòng tiền của bạn không cần viết DAX rườm rà.

sql
CREATE OR REPLACE VIEW `minpyws.techstore_analytics.vw_cashflow_daily` AS
WITH daily_revenue AS (
    -- Bước 1: Tính tổng doanh số từ các đơn hàng thành công mỗi ngày
    SELECT
        DATE(order_date) AS report_date,
        SUM(total_vnd) AS sales_revenue_vnd,
        COUNT(*) AS order_count
    FROM `minpyws.techstore_analytics.fact_orders`
    WHERE status NOT IN ('cancelled', 'refunded')
    GROUP BY DATE(order_date)
),
daily_payments AS (
    -- Bước 2: Tính tổng tiền thanh toán online (MoMo, ZaloPay, PayPal) mỗi ngày
    SELECT
        DATE(payment_date) AS report_date,
        SUM(CASE WHEN payment_status = 'success' THEN amount_vnd ELSE 0 END) AS payments_received_vnd,
        COUNT(CASE WHEN payment_status = 'success' THEN 1 END) AS successful_payments,
        COUNT(CASE WHEN payment_status != 'success' THEN 1 END) AS failed_payments
    FROM `minpyws.techstore_analytics.fact_payments`
    GROUP BY DATE(payment_date)
),
daily_bank AS (
    -- Bước 3: Tính tổng tiền ra/vào trên sao kê ngân hàng Mercury mỗi ngày
    SELECT
        DATE(transaction_date) AS report_date,
        SUM(CASE WHEN amount_vnd > 0 THEN amount_vnd ELSE 0 END) AS bank_inflow_vnd,
        SUM(CASE WHEN amount_vnd < 0 THEN ABS(amount_vnd) ELSE 0 END) AS bank_outflow_vnd
    FROM `minpyws.techstore_analytics.fact_bank_transactions`
    WHERE status IN ('posted', 'pending')
    GROUP BY DATE(transaction_date)
)
-- Bước 4: Ghép 3 bảng con lại với nhau tính dòng tiền thực tế
SELECT
    COALESCE(r.report_date, p.report_date, b.report_date) AS report_date,
    COALESCE(r.sales_revenue_vnd, 0) AS sales_revenue_vnd,
    COALESCE(r.order_count, 0) AS order_count,
    COALESCE(p.payments_received_vnd, 0) AS payments_received_vnd,
    COALESCE(p.successful_payments, 0) AS successful_payments,
    COALESCE(p.failed_payments, 0) AS failed_payments,
    COALESCE(b.bank_inflow_vnd, 0) AS bank_inflow_vnd,
    COALESCE(b.bank_outflow_vnd, 0) AS bank_outflow_vnd,
    COALESCE(r.sales_revenue_vnd, 0) - COALESCE(b.bank_outflow_vnd, 0) AS net_cashflow_vnd
FROM daily_revenue r
FULL OUTER JOIN daily_payments p ON r.report_date = p.report_date
FULL OUTER JOIN daily_bank b ON COALESCE(r.report_date, p.report_date) = b.report_date
ORDER BY report_date
2. View: Trạng thái Thanh toán (vw_payment_status)
View này nối fact_orders với dim_customers và thông tin chi tiết thanh toán fact_payments. Đặc biệt, nó sử dụng hàm CASE WHEN để tự động phân loại đơn hàng thành Paid, Failed, Pending, và Overdue (nếu quá hạn 30 ngày). View này cực kỳ hữu ích để làm báo cáo công nợ.

sql
CREATE OR REPLACE VIEW `minpyws.techstore_analytics.vw_payment_status` AS
SELECT
    o.order_key,
    o.order_id,
    o.order_number,
    o.customer_id,
    c.full_name AS customer_name,
    c.email AS customer_email,
    o.order_date,
    o.channel,
    o.total_vnd,
    o.payment_status AS order_payment_status,
    o.payment_gateway,
    p.payment_key,
    p.payment_date,
    p.payment_status AS gateway_payment_status,
    p.amount_vnd AS paid_amount_vnd,
    
    -- TỰ ĐỘNG PHÂN LOẠI TRẠNG THÁI THANH TOÁN KHÁCH HÀNG:
    CASE
        WHEN p.payment_status = 'success' THEN 'Paid'
        WHEN p.payment_status = 'failed' THEN 'Failed'
        WHEN p.payment_date IS NULL AND TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), o.order_date, DAY) > 30 THEN 'Overdue'
        WHEN p.payment_date IS NULL THEN 'Pending'
        ELSE 'Unknown'
    END AS payment_status_category,
    
    CASE
        WHEN p.payment_date IS NOT NULL THEN
            TIMESTAMP_DIFF(p.payment_date, o.order_date, HOUR)
        ELSE NULL
    END AS payment_delay_hours,
    
    -- TÍNH SỐ TIỀN CÒN NỢ (outstanding_amount_vnd)
    CASE
        WHEN p.payment_status != 'success' OR p.payment_status IS NULL THEN
            o.total_vnd - COALESCE(p.amount_vnd, 0)
        ELSE 0
    END AS outstanding_amount_vnd,
    
    -- TÍNH SỐ NGÀY QUÁ HẠN (days_overdue)
    CASE
        WHEN p.payment_date IS NULL AND TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), o.order_date, DAY) > 30 THEN
            TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), o.order_date, DAY)
        ELSE 0
    END AS days_overdue
FROM `minpyws.techstore_analytics.fact_orders` o
LEFT JOIN `minpyws.techstore_analytics.dim_customers` c
    ON o.customer_id = c.customer_id
LEFT JOIN `minpyws.techstore_analytics.fact_payments` p
    ON o.transaction_id = p.transaction_id
3. View: Hành trình Khách hàng (vw_customer_journey)
View này phức tạp nhất vì nó nối bảng sự kiện lướt web/thêm giỏ hàng (fact_cart_events) với thông tin khách hàng (dim_customers) và cả các giao dịch mua hàng (fact_orders). Nó cung cấp cho bạn biết số lượt tương tác (touchpoint sequence) và thời gian mất bao lâu từ lúc họ xem sp tới khi họ mua hàng (days_to_first_purchase).

sql
CREATE OR REPLACE VIEW `minpyws.techstore_analytics.vw_customer_journey` AS
WITH customer_events AS (
    -- Bước 1: Tính thứ tự sắp xếp thời gian các sự kiện thêm giỏ hàng, xem SP
    SELECT
        ce.customer_id,
        ce.session_id,
        ce.event_type,
        ce.event_timestamp,
        ce.product_id,
        ce.source AS traffic_source,
        ce.device,
        ce.utm_source,
        ce.utm_campaign,
        ROW_NUMBER() OVER (
            PARTITION BY ce.customer_id
            ORDER BY ce.event_timestamp
        ) AS touchpoint_sequence
    FROM `minpyws.techstore_analytics.fact_cart_events` ce
    WHERE ce.customer_id IS NOT NULL
),
customer_purchases AS (
    -- Bước 2: Tìm ngày mua hàng đầu tiên và rà soát tổng số tiền khách từng mua
    SELECT
        customer_id,
        MIN(order_date) AS first_purchase_date,
        COUNT(*) AS total_purchases,
        SUM(total_vnd) AS total_revenue_vnd
    FROM `minpyws.techstore_analytics.fact_orders`
    WHERE customer_id IS NOT NULL
    GROUP BY customer_id
)
-- Bước 3: Mix lại toàn bộ các thông tin làm Dashboard Hành trình
SELECT
    e.customer_id,
    c.full_name,
    c.customer_segment,
    e.session_id,
    e.event_type,
    e.event_timestamp,
    e.touchpoint_sequence,
    e.product_id,
    e.traffic_source,
    e.device,
    e.utm_source,
    e.utm_campaign,
    p.first_purchase_date,
    p.total_purchases,
    p.total_revenue_vnd,
    
    -- TÍNH THỜI GIAN MẤT BAO LÂU (TÍNH THEO NGÀY) TỪ LÚC TƯƠNG TÁC ĐẦU TIÊN TỚI ĐƠN ĐẦU
    TIMESTAMP_DIFF(p.first_purchase_date, MIN(e.event_timestamp) OVER (PARTITION BY e.customer_id), DAY) AS days_to_first_purchase
FROM customer_events e
LEFT JOIN `minpyws.techstore_analytics.dim_customers` c
    ON e.customer_id = c.customer_id
LEFT JOIN customer_purchases p
    ON e.customer_id = p.customer_id
Tại sao lại chọn tạo View?
Quy trình bạn vừa setup được gọi là làm theo hướng ELT (Extract - Load - Transform). Mã Python của chúng ta tạo Dim và Fact và Load lên BigQuery. Bản chất của việc này là để tối ưu tốc độ trực quan hóa dữ liệu.

View là gì?
Hiểu đơn giản nhất
Loại	Lưu trữ data?	Ví dụ
Table (dim, fact)	✅ Có — data thật nằm trên ổ cứng	dim_customers, fact_orders
View	❌ Không — chỉ là câu SQL được lưu sẵn	vw_cashflow_daily
View = một câu SELECT query được đặt tên và lưu lại, mỗi lần bạn mở view nó sẽ chạy lại câu query đó trên data thật.

Ví dụ cụ thể
Bảng vw_cashflow_daily thực chất là câu SQL này được lưu sẵn:

sql
-- Mỗi lần mở view, BigQuery tự chạy câu này:
SELECT
    DATE(order_date) AS report_date,
    SUM(total_vnd) AS sales_revenue_vnd,     -- Lấy từ fact_orders
    ...
    SUM(amount_vnd) AS payments_received_vnd  -- Lấy từ fact_payments
    ...
    SUM(amount_vnd) AS bank_inflow_vnd        -- Lấy từ fact_bank_transactions
FROM fact_orders
JOIN fact_payments ...
JOIN fact_bank_transactions ...
GROUP BY report_date

1. Nhóm chỉ số Bán hàng & Đơn hàng (Sales & Orders)
Nhóm này thao tác chủ yếu trên bảng fact_orders và fact_order_items để lấy ra bức tranh tổng quan về doanh thu.

dax
// 1. Tổng doanh thu
Total Revenue = SUM('fact_orders'[total_amount]) 
// (Nếu doanh thu không nằm trong bảng orders, bạn có thể tính bằng: SUMX('fact_order_items', 'fact_order_items'[price] * 'fact_order_items'[quantity]))
// 2. Tổng số lượng đơn hàng
Total Orders = DISTINCTCOUNT('fact_orders'[order_id])
// 3. Số lượng khách hàng đã mua
Total distinct Customers = DISTINCTCOUNT('fact_orders'[customer_id])
// 4. Giá trị trung bình của 1 đơn hàng (AOV - Average Order Value)
AOV = DIVIDE([Total Revenue], [Total Orders], 0)
// 5. Đơn hàng hoàn thành (Giả sử cột status lưu trạng thái "Delivered" hoặc "Completed")
Completed Orders = CALCULATE([Total Orders], 'fact_orders'[status] = "completed")
2. Nhóm chỉ số Thanh toán & Dòng tiền (Payment & Cashflow)
Sử dụng trên bảng fact_payments hoặc fact_bank_transactions để phân tích dòng tiền vào/ra và các lỗi thanh toán.

dax
// 6. Tổng số tiền thanh toán
Total Payment Amount = SUM('fact_payments'[amount])
// 7. Số lượng giao dịch thành công
Successful Transactions Count = CALCULATE(COUNTROWS('fact_payments'), 'fact_payments'[payment_status] = "success")
// 8. Tỉ lệ thanh toán thành công (%)
Payment Success Rate = DIVIDE([Successful Transactions Count], COUNTROWS('fact_payments'), 0)
// 9. Tổng tiền hoàn (Refund)
Total Refund Amount = CALCULATE(SUM('fact_payments'[amount]), 'fact_payments'[payment_status] = "refunded")
3. Nhóm phân tích hành vi khách hàng (Customer Journey)
Nếu bạn không dùng bảng View sẵn có mà muốn tự phân tích sâu hành vi từ bảng fact_cart_events.

dax
// 10. Số lượt xem sản phẩm
Total Views = CALCULATE(COUNTROWS('fact_cart_events'), 'fact_cart_events'[event_type] = "view_item")
// 11. Số lượt Thêm vào giỏ hàng
Total Add to Cart = CALCULATE(COUNTROWS('fact_cart_events'), 'fact_cart_events'[event_type] = "add_to_cart")
// 12. Tỉ lệ bỏ giỏ hàng (Cart Abandonment Rate)
Cart Abandonment Rate = 
VAR ConversionToOrder = DIVIDE([Total Orders], [Total Add to Cart], 0)
RETURN 1 - ConversionToOrder
4. Nhóm Time Intelligence (So sánh thời gian YTD, MoM)
IMPORTANT

Để các hàm Time Intelligence hoạt động đúng, bạn bắt buộc phải có mối quan hệ (Relationship) giữa bảng dim_date (cột Date) và fact_orders (cột Order Date), đồng thời set bảng dim_date là "Mark as Date Table" trong Power BI.

dax
// 13. Doanh thu lũy kế từ đầu năm đến hiện tại (Year-To-Date)
Revenue YTD = TOTALYTD([Total Revenue], 'dim_date'[date])
// 14. Doanh thu tháng trước
Revenue Previous Month = CALCULATE([Total Revenue], PREVIOUSMONTH('dim_date'[date]))
// 15. Tăng trưởng Doanh thu theo tháng so với tháng trước (Month-over-Month %)
